% The card
\newcommand{\card}[3]{\begin{tikzpicture}[overlay,remember picture]
    \draw [line width=2mm]
        ($ (current page.north west) + (1mm,-1mm) $)
        rectangle
        ($ (current page.south east) + (-1mm,1mm) $);
    \draw [line width=3pt, dotted] % (10mm, -10mm) -- (60mm, -10mm );
        ($ (current page.north west) + (6mm, -13mm) $)
        --
        ($ (current page.north east) + (-6mm, -13mm) $);
\end{tikzpicture}
\setlength\unitlength{1mm}%
\begin{picture}(0,0)
\put(84,-2){#2}%
\put(0,-1){\cardtitle{#1}}
\end{picture}
\vspace{5mm}

\cardtext{#3}
}
